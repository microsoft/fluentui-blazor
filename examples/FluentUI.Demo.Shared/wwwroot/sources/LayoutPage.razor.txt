<h1>Layout</h1>

<p>
    The <code>Layout</code> component is a container for other components. It is used to create a layout for a page.
</p>
<p>
    The <code>Layout</code> component has a <code>ChildContent</code> parameter that can be used to add other components to the layout.
</p>

<ApiDocumentation Component="typeof(Layout)" />

<h2>Example</h2>
<DemoSection Component="typeof(Layout)" Title="Layout" CollocatedFiles="@(new[] {"cs", "css"})" ComponentParameters="@(new Dictionary<string, object>(){ {"ChildContent",(RenderFragment)(builder => builder.AddMarkupContent(1,"<div style=\"display: flex;\">This is some content displayed in a Layout</div>")) } })">
    <ChildContent>
        <CodeSnippet>
&lt;Layout&gt;
    This is some content displayed in a Layout
&lt;/Layout&gt;
        </CodeSnippet>
        <p>Renders as:</p>
    </ChildContent>
</DemoSection>
