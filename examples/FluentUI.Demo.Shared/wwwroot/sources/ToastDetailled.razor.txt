@inject IToastService ToastService

<p>Click on this button to open a Toast. The toast is created by specifying an intent, a message, an  action and a rando timeout between  and 15 seconds.</p>

<FluentButton @onclick="@OpenToast" Appearance="Appearance.Accent">Open</FluentButton>

@code
{
    int counter = 1;
    void OpenToast()
    {
        Random rnd = new();

        var intent = Enum.GetValues<ToastIntent>()[rnd.Next(10)];
        var message = $"Simple Toast #{counter++}";

        ToastService.ShowToast(
            intent, 
            message, 
            action =>
            { 
                action.Text = "Visit site"; action.OnClick = () => DemoLogger.WriteLine($"Action clicked"); 
            },
            settings => 
            { 
                settings.Timeout = rnd.Next(15); 
            });
    }
}
